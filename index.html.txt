        let currentCardIndex = 0;
        let isFlipped = false;

        const flashcardElement = document.getElementById('flashcard');
        const flashcardFront = document.getElementById('flashcard-front');
        const flashcardBack = document.getElementById('flashcard-back');
        const toggleButton = document.getElementById('toggleAnswer');
        const prevButton = document.getElementById('prevCard');
        const nextButton = document.getElementById('nextCard');
        const currentCardNumSpan = document = document.getElementById('currentCardNum');
        const totalCardsSpan = document.getElementById('totalCards');

        /**
         * Displays the current flashcard (question side) based on the currentCardIndex.
         * Updates the card number display.
         */
        function displayCard() {
            // Set the question on the front of the card
            flashcardFront.innerHTML = `<p class="text-center">${flashcards[currentCardIndex].question}</p>`;
            // Set the answer on the back of the card
            flashcardBack.innerHTML = `<p class="text-center">${flashcards[currentCardIndex].answer}</p>`;

            // Update card counter
            currentCardNumSpan.textContent = currentCardIndex + 1;
            totalCardsSpan.textContent = flashcards.length;

            // Ensure card is on the front side when a new question is displayed
            if (isFlipped) {
                flashcardElement.classList.remove('flipped');
                isFlipped = false;
                toggleButton.textContent = 'Show Answer';
            }
        }

        /**
         * Toggles the flip state of the flashcard, showing either the question or the answer.
         */
        function toggleAnswer() {
            isFlipped = !isFlipped;
            flashcardElement.classList.toggle('flipped', isFlipped);
            toggleButton.textContent = isFlipped ? 'Hide Answer' : 'Show Answer';
        }

        /**
         * Navigates to the next flashcard in the array.
         * Wraps around to the beginning if at the end.
         */
        function nextCard() {
            currentCardIndex = (currentCardIndex + 1) % flashcards.length;
            displayCard();
        }

        /**
         * Navigates to the previous flashcard in the array.
         * Wraps around to the end if at the beginning.
         */
        function prevCard() {
            currentCardIndex = (currentCardIndex - 1 + flashcards.length) % flashcards.length;
            displayCard();
        }

        // Add event listeners to buttons
        toggleButton.addEventListener('click', toggleAnswer);
        prevButton.addEventListener('click', prevCard);
        nextButton.addEventListener('click', nextCard);

        // Initial display of the first card
        document.addEventListener('DOMContentLoaded', displayCard);

                        console.log('Service Worker registered successfully:', registration);
                    })
                    .catch(error => {
                        console.error('Service Worker registration failed:', error);
                    });
            });
